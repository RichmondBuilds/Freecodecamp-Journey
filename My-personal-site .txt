// ── CURSOR ──
const dot = document.getElementById('c-dot');
const ring = document.getElementById('c-ring');
let mx=0,my=0,rx=0,ry=0;
document.addEventListener('mousemove',e=>{mx=e.clientX;my=e.clientY;});
function animCursor(){
  rx+=(mx-rx)*.13; ry+=(my-ry)*.13;
  dot.style.left=mx+'px'; dot.style.top=my+'px';
  ring.style.left=rx+'px'; ring.style.top=ry+'px';
  requestAnimationFrame(animCursor);
}
animCursor();
document.querySelectorAll('a,button,input,select,textarea,.svc-card,.b-card,.deal-type').forEach(el=>{
  el.addEventListener('mouseenter',()=>document.body.classList.add('cursor-grow'));
  el.addEventListener('mouseleave',()=>document.body.classList.remove('cursor-grow'));
});

// ── PARALLAX ──
const p1=document.getElementById('parallax-1');
const p2=document.getElementById('parallax-2');
document.addEventListener('mousemove',e=>{
  const xp=(e.clientX/window.innerWidth-.5)*20;
  const yp=(e.clientY/window.innerHeight-.5)*20;
  p1.style.transform=`translate(${xp*.5}px,${yp*.5}px)`;
  p2.style.transform=`translate(${xp*.3}px,${yp*.3}px)`;
});

// scroll parallax for depth cards
window.addEventListener('scroll',()=>{
  const y=window.scrollY;
  const dc1=document.querySelector('.dc1');
  const dc2=document.querySelector('.dc2');
  const dc3=document.querySelector('.dc3');
  if(dc1){ dc1.style.transform=`translateY(${y*.03}px)`; }
  if(dc2){ dc2.style.transform=`translateY(${y*.05}px)`; }
  if(dc3){ dc3.style.transform=`translateY(${y*.08}px)`; }
});

// ── PROJECT COUNTER ──
function countUp(el, target, duration){
  let start = 0;
  const step = target / (duration / 16);
  const timer = setInterval(() => {
    start += step;
    if (start >= target) { start = target; clearInterval(timer); }
    el.textContent = Math.floor(start) + '+';
  }, 16);
}
const pcEl = document.getElementById('proj-count');
const pcObs = new IntersectionObserver(entries => {
  entries.forEach(e => {
    if (e.isIntersecting) { countUp(pcEl, 140, 2500); pcObs.disconnect(); }
  });
}, { threshold: 0.3 });
if (pcEl) pcObs.observe(pcEl);

// ── SCROLL REVEAL ──
const srObs=new IntersectionObserver(entries=>{
  entries.forEach(e=>{if(e.isIntersecting)e.target.classList.add('show');});
},{threshold:.1});
document.querySelectorAll('.sr').forEach(el=>srObs.observe(el));

// ── FORM SUBMIT ──
function submitDeal(){
  const name=document.getElementById('f-name').value.trim();
  const email=document.getElementById('f-email').value.trim();
  const type=document.getElementById('f-type').value;
  const msg=document.getElementById('f-msg').value.trim();
  if(!name||!email||!email.includes('@')||!type||!msg){
    // shake empty fields
    [['f-name',name],['f-email',email],['f-msg',msg]].forEach(([id,val])=>{
      if(!val){
        const el=document.getElementById(id);
        el.style.borderColor='var(--accent2)';
        setTimeout(()=>el.style.borderColor='',1500);
      }
    });
    return;
  }
  document.getElementById('deal-form-wrap').style.display='none';
  document.getElementById('form-sent').style.display='block';
}

// enter key on inputs
['f-name','f-email','f-budget'].forEach(id=>{
  document.getElementById(id)?.addEventListener('keydown',e=>{if(e.key==='Enter')submitDeal();});
});
