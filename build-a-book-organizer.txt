function sortByYear(a, b) {
  if (a.releaseYear < b.releaseYear) {
    return -1;
  } else if (a.releaseYear > b.releaseYear) {
    return 1;
  } else {
    return 0;
  }
}
const filteredBooks = books.filter(books => books.releaseYear <= 1950);
filteredBooks.sort(sortByYear);
