** start of script.js **

const fullName = document.getElementById("full-name");
const email = document.getElementById("email");
const orderNo = document.getElementById("order-no");
const productCode = document.getElementById("product-code");
const quantity = document.getElementById("quantity");
const complaintsGroup = document.getElementById("complaints-group");
const complaintDescription = document.getElementById("complaint-description");
const solutionsGroup = document.getElementById("solutions-group");
const solutionDescription = document.getElementById("solution-description");
const checkboxes = complaintsGroup.querySelectorAll("input[type='checkbox']");
const otherComplaint = document.querySelector("#complaints-group input[value='other']");
const otherSolution = document.querySelector("#solutions-group input[value='other']");
const radioButtons = solutionsGroup.querySelectorAll("input[type='radio']");
const form = document.querySelector("form");

function validateForm() {
  const isOtherComplaintChecked = otherComplaint && otherComplaint.checked;
  const isOtherSolutionSelected = otherSolution && otherSolution.checked;

  return {
    "full-name": fullName.value.trim() !== "",
    "email": /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email.value),
    "order-no": /^2024\d{6}$/.test(orderNo.value),
    "product-code": /^[a-zA-Z]{2}\d{2}-[a-zA-Z]\d{3}-[a-zA-Z]{2}\d$/.test(productCode.value),
    "quantity": /^[1-9]\d*$/.test(quantity.value),
    "complaints-group": Array.from(checkboxes).some((cb) => cb.checked),
    "complaint-description": isOtherComplaintChecked ? complaintDescription.value.trim().length >= 20 : true,
    "solutions-group": Array.from(radioButtons).some((radio) => radio.checked),
    "solution-description": isOtherSolutionSelected ? solutionDescription.value.trim().length >= 20 : true
  };
}

function isValid(validationObj) {
  return Object.values(validationObj).every((value) => value === true);
}

function highlightFields(validationObj) {
  const fieldMap = {
    "full-name": fullName,
    "email": email,
    "order-no": orderNo,
    "product-code": productCode,
    "quantity": quantity,
    "complaints-group": complaintsGroup,
    "complaint-description": complaintDescription,
    "solutions-group": solutionsGroup,
    "solution-description": solutionDescription
  };

  for (const key in validationObj) {
    const el = fieldMap[key];
    el.style.borderColor = validationObj[key] ? "green" : "red";
  }
}

form.addEventListener("submit", (e) => {
  e.preventDefault();
  const validationObj = validateForm();
  highlightFields(validationObj);
  if (isValid(validationObj)) {
    alert("Form submitted successfully!");
  }
});

const allFields = [fullName, email, orderNo, productCode, quantity, complaintDescription, solutionDescription];

allFields.forEach((field) => {
  field.addEventListener("change", () => {
    const validationObj = validateForm();
    highlightFields(validationObj);
  });
});

checkboxes.forEach((cb) => {
  cb.addEventListener("change", () => {
    const validationObj = validateForm();
    complaintsGroup.style.borderColor = validationObj["complaints-group"] ? "green" : "red";
  });
});

radioButtons.forEach((radio) => {
  radio.addEventListener("change", () => {
    const validationObj = validateForm();
    solutionsGroup.style.borderColor = validationObj["solutions-group"] ? "green" : "red";
  });
});
