function convertHTML(str) {
let result = "";
const htmlEntities = {
  "&": "&amp;",
  "<": "&lt;",
  ">": "&gt;",
  '"': "&quot;",
  "'": "&apos;"
};
for (const char of str) {
  if (htmlEntities[char]) {
    result += htmlEntities[char];
  } else {
    result += char;
  }
}
return result;

}
