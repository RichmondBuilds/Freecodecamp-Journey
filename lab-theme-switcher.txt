** start of script.js **

const themes = [
  {
    name: "richmondbuilds",
    message: "Building different. Cold like that, Different breed 🥶"
  },
  {
    name: "foreigner",
    message: "Abroad mentality, Ghana roots 🌍"
  }
];

const themeSwitcher = document.querySelector("#theme-switcher-button");
const themeDropdown = document.querySelector("#theme-dropdown");

themeSwitcher.addEventListener("click", () => {
  themeDropdown.toggleAttribute("hidden");
  if (!themeDropdown.hasAttribute("hidden")) {
    themeSwitcher.setAttribute("aria-expanded", "true");
  } else {
    themeSwitcher.setAttribute("aria-expanded", "false");
  }
});

const ariaLive = document.querySelector("[aria-live='polite']");
const liEls = document.querySelectorAll("li");

liEls.forEach((li) => {
  li.addEventListener("click", () => {
    const selectedName = li.textContent.trim().toLowerCase();
    const selectedTheme = themes.find((theme) => theme.name === selectedName);
    document.body.className = `theme-${selectedName}`;
    ariaLive.textContent = selectedTheme.message;
  });
});
